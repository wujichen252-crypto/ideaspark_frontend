# 开发模式启动
npm run dev

# 生产环境打包（包含类型检查）
npm run build

# 代码检查
npm run lint

# 自动修复代码问题
npm run lint:fix

# 代码格式化
npm run format

# TypeScript 类型检查
npm run typecheck

# 预览生产构建
npm run preview

# 运行测试
npm run test

# 运行测试（一次性）
npm run test:run

# 测试覆盖率
npm run test:coverage

# 1. 安装依赖（如果还没安装）
npm install

# 2. 启动开发服务器
npm run dev

# 3. 打包生产版本
npm run build
